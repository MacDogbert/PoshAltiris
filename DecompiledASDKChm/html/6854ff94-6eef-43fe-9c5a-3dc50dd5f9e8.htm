Sample input XML to apply image multicast settings.

CopyXML
<DeploymentItem>
  <ItemType>Settings</ItemType>
  <ItemName>ImageMulticast</ItemName>
  <ItemAction>apply</ItemAction>
  <Item>
    <MultiCastSettings>
      <GlobalSettings>
        <Threshold>11</Threshold>
        <BroadcastLimit>10</BroadcastLimit>
        <TimeOut>11</TimeOut>
        <IPAddress_from>224.2.0.5</IPAddress_from>
        <IPAddress_to>224.2.0.20</IPAddress_to>
        <Port_from>409</Port_from>
        <Port_to>402</Port_to>
      </GlobalSettings>
    </MultiCastSettings>
  </Item>
</DeploymentItem>
